            <div class="artist-name">Bin Youn</div>

            <p>This work is inspired by the East Asian tradition of scholars gathering in nature to compose and exchange poetry—a practice known as <em>pungnyu</em> (풍류) in Korea. The word itself reveals its essence: <em>pung</em> (풍/風) means wind, and <em>ryu</em> (류/流) means flow—together suggesting the way art and life should move like wind and water, naturally and without force.</p>

            

            <p>These <em>seonbi</em> (scholar-artist) believed that true culture arose through spontaneous exchange, where one person's verse would inspire another's response, flowing from voice to voice like wind through leaves, creating layers of meaning that no single voice could achieve.</p>

            

            <p>Within RMIT Vietnam's global community, I transpose this analog tradition into our digital age. Just as <em>pungnyu</em> scholars found infinite expression within nature's patterns, the space beyond 1 and 0 contains everything that makes us human.</p>

            

            <p>This digital gathering invites our community to contribute fragments throughout the exhibition period. Each submission becomes part of a larger consciousness, creating the same spontaneous harmony that <em>pungnyu</em> sought centuries ago.</p>

            

            <p class="statement-note">The final collective poem will be compiled and released after the exhibition closes on September 18, 2025.</p>

        function submitWord() {

            const word = input.value.trim();

            

            if (word) {

                console.log('New entry:', entry);

            }

        }

        

        function sendToGoogleSheets(word) {

            const formUrl = `https://docs.google.com/forms/d/e/${GOOGLE_FORM_ID}/formResponse`;

            const formData = new FormData();

            formData.append(GOOGLE_FORM_ENTRY_ID, word);

            

            // Send data to Google Form (no CORS issues with forms)

            fetch(formUrl, {

                method: 'POST',

                mode: 'no-cors',

                body: formData

            }).then(() => {

                console.log('Submitted to Google Sheets:', word);

            }).catch((error) => {

                console.error('Error submitting to Google Sheets:', error);

                // Still works even if this shows an error

            });

        }

        

        function addWordToDisplay(word) {

            const display = document.getElementById('wordsDisplay');

            

            // Add new word

            displayWords.push(word);

            

            // Keep only recent words

            if (displayWords.length > MAX_DISPLAY) {

                displayWords.shift();

            }

            

            // Rebuild display

            display.innerHTML = '';

            displayWords.forEach(w => {

                const wordElement = document.createElement('span');

                wordElement.className = 'word';

                wordElement.textContent = w;

                display.appendChild(wordElement);

            });

        }

        

        function updateDisplay() {

            // Show recent words

            const recent = words.slice(-MAX_DISPLAY).map(entry => entry.word);

            displayWords = recent;

            

            const display = document.getElementById('wordsDisplay');

            display.innerHTML = '';

            

            recent.forEach(word => {

                const wordElement = document.createElement('span');

                wordElement.className = 'word';

                wordElement.textContent = word;

                display.appendChild(wordElement);

            });

        }

        

        function updateCounter() {

            document.getElementById('count').textContent = words.length;

        }

            const dataStr = JSON.stringify(words, null, 2);

            const dataUri = 'data:application/json;charset=utf-8,'+ encodeURIComponent(dataStr);

            

            const exportFileDefaultName = `collective-poem-${new Date().toISOString()}.json`;

            

            const linkElement = document.createElement('a');

            linkElement.setAttribute('href', dataUri);

            linkElement.setAttribute('download', exportFileDefaultName);

            linkElement.click();

        }
